DDos script with autoinstall and compile !

##Ubuntu & Debian

	apt-get install gcc
	gcc [script-name].c -pthread -o nameofscript
## CentOS
yum -y update
yum install gcc
gcc [script-name].c -pthread -o nameofscript

git clone 
sh ./putinsource.sh -all
